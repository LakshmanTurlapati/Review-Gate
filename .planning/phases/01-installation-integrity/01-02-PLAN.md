  <verify>python3 -m py_compile V2/update_mcp_config.py && python3 - <<'PY'
import json
import pathlib
import subprocess
import tempfile

with tempfile.TemporaryDirectory() as td:
    config_path = pathlib.Path(td) / 'mcp.json'
    config_path.write_text(json.dumps({
        'mcpServers': {
            'other-server': {'command': 'node', 'args': ['server.js']},
            'review-gate-v2': {'command': 'old-python'}
        }
    }, indent=2), encoding='utf-8')

    subprocess.run([
        'python3', 'V2/update_mcp_config.py', 'merge',
        '--config', str(config_path),
        '--template', 'V2/mcp.json',
        '--server-name', 'review-gate-v2',
        '--install-dir', 'C:/Users/Test/cursor-extensions/review-gate-v2',
        '--python-cmd', 'C:/Users/Test/cursor-extensions/review-gate-v2/venv/Scripts/python.exe'
    ], check=True)

    merged = json.loads(config_path.read_text(encoding='utf-8'))
    assert 'other-server' in merged['mcpServers']
    assert merged['mcpServers']['review-gate-v2']['command'].endswith('venv/Scripts/python.exe')

    subprocess.run([
        'python3', 'V2/update_mcp_config.py', 'remove',
        '--config', str(config_path),
        '--server-name', 'review-gate-v2'
    ], check=True)

    removed = json.loads(config_path.read_text(encoding='utf-8'))
    assert 'other-server' in removed['mcpServers']
    assert 'review-gate-v2' not in removed['mcpServers']

print('ok')
PY</verify>
